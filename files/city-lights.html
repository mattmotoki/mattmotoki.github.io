import numpy as np
import matplotlib.pyplot as plt

def hex2rgb(hex_value):
    h = hex_value.strip('#')
    rgb = tuple(int(h[i:i+2], 16)/255 for i in (0, 2, 4))
    return rgb

yellow_spectrum = [
    '#fffdf2', '#fffcea', '#fefbe2', '#fefada', '#fef9d2', '#fef8c9',
    '#fef7c1', '#fef6b9', '#fef5b1', '#fff4a9', '#fff3a1', '#fff299',
    '#fff190', '#fff088', '#ffef80', '#ffee78', '#ffed70', '#ffec68',
    '#ffeb5f', '#ffea57', '#ffe84f', '#ffe747', '#ffe63f', '#ffe537',
    '#ffe42e', '#ffe326', '#ffe21e', '#ffe116', '#ffe00e', '#ffdf06',
    '#fcdd00', '#f4d600', '#ecce00', '#e4c700', '#dcc000', '#d4b900',
    '#ccb200', '#c3ab00', '#bba400', '#b39c00', '#ab9500', '#a38e00',
    '#9b8700', '#928000', '#8a7900', '#827200', '#7a6b00', '#726300',
    '#6a5c00', '#615500', '#594e00', '#514700', '#494000', '#413900',
]

yellow_spectrum = np.array([hex2rgb(x) for x in yellow_spectrum])

np.random.seed(42)
plt.figure(facecolor='none')
ax = plt.gca()

def get_color(n):
    x = np.clip(5*np.random.randn(n) + 40, 0, 50)
    return yellow_spectrum[x.astype(np.int64)]

def draw_building(ax, offset, h, w, wsize=0.5):
    ax.bar(offset, h, width=w, color='#313131', edgecolor='#2C251F')
    n_windows = max(np.random.randint(int(w**0.5*h)), 10)
    x_pos = np.random.randint(0, w-1, n_windows).astype(np.float32)
    y_pos = np.random.randint(0, h-1, n_windows).astype(np.float32)
    x_pos += offset + (2 - wsize - w)/2
    y_pos += (2 - wsize)/2    
    window_colors = get_color(n_windows)
    
    for x, y, w_color in zip(x_pos, y_pos, window_colors):
        rect = plt.Rectangle((x, y), wsize, wsize, color=w_color)
        ax.add_patch(rect)

n_buildings = np.random.randint(25, 50)
positions = np.random.rand(n_buildings) * 100
heights = np.random.randint(5, 55, n_buildings)
widths = np.random.randint(3, 10, n_buildings)

for x, h, w in zip(positions, heights, widths):
    draw_building(ax, x, h, w)

ax.get_xaxis().set_visible(False)
ax.get_yaxis().set_visible(False)
ax.set_xlim(-10, 110)
ax.set_ylim(0, 60)
ax.set_aspect('equal', 'box')
ax.set_axis_off()

plt.tight_layout()
plt.show()
